  // ----------- Coin Flip Animation -----------

  function flipCoin() {

    const coin = document.getElementById("flip-coin");

    coin.classList.add("flipping");

    document.getElementById("feedback").textContent = "Flipping...";

      coin.classList.remove("flipping");

  }
